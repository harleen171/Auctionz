// ========== TESTING WITH JAVASCRIPT ==========

const API_URL = 'http://localhost:3000/api';
let authToken = null;

// Helper function
async function apiCall(endpoint, method = 'GET', body = null) {
  const options = {
    method,
    headers: {
      'Content-Type': 'application/json',
    }
  };

  if (authToken) {
    options.headers.Authorization = `Bearer ${authToken}`;
  }

  if (body) {
    options.body = JSON.stringify(body);
  }

  const response = await fetch(API_URL + endpoint, options);
  return response.json();
}

// Test signup
async function testSignup() {
  const result = await apiCall('/auth/signup', 'POST', {
    fullName: 'Test User',
    username: 'testuser',
    email: 'test@example.com',
    password: 'Password123'
  });
  
  if (result.success) {
    authToken = result.token;
    console.log('✅ Signup successful');
  } else {
    console.log('❌ Signup failed:', result.message);
  }
}

// Test login
async function testLogin() {
  const result = await apiCall('/auth/login', 'POST', {
    email: 'test@example.com',
    password: 'Password123'
  });
  
  if (result.success) {
    authToken = result.token;
    console.log('✅ Login successful, token:', authToken.substring(0, 20) + '...');
  } else {
    console.log('❌ Login failed:', result.message);
  }
}

// Test protected route
async function testProtected() {
  if (!authToken) {
    console.log('❌ No token available. Login first!');
    return;
  }

  const result = await apiCall('/protected');
  console.log('✅ Protected route:', result);
}

// Test create auction
async function testCreateAuction() {
  if (!authToken) {
    console.log('❌ No token available. Login first!');
    return;
  }

  const result = await apiCall('/auctions', 'POST', {
    title: 'Test Auction',
    description: 'This is a test auction',
    category: 'Electronics',
    startingPrice: 100,
    startTime: new Date().toISOString(),
    endTime: new Date(Date.now() + 7*24*60*60*1000).toISOString(),
    condition: 'like-new',
    location: 'Test Location'
  });

  if (result.success) {
    console.log('✅ Auction created:', result.auction._id);
  } else {
    console.log('❌ Failed:', result.message);
  }
}

// Test get auctions
async function testGetAuctions() {
  const result = await apiCall('/auctions?page=1&limit=5');
  console.log('✅ Auctions:', result.auctions.length, 'found');
  console.log('📊 Total:', result.pagination.total);
}

// Run tests
async function runTests() {
  console.log('🧪 Starting API tests...\n');
  
  await testSignup();
  await testProtected();
  await testCreateAuction();
  await testGetAuctions();
  
  console.log('\n✅ Tests complete!');
}

/*
Error: "Email already exists"
Solution: Use a different email address

Error: "Unauthorized - Please login first"
Solution: Make sure Authorization header is set correctly
         Format: "Bearer {token}" (note the space)

Error: "Invalid or expired token"
Solution: Your JWT token has expired, login again

Error: "Password must be at least 6 characters"
Solution: Password requirements:
         - At least 6 characters
         - At least one uppercase letter
         - At least one number

Error: "End time must be after start time"
Solution: Make sure endTime is later than startTime
         Example: now + 7 days

Error: "Cannot update active auction"
Solution: Only inactive auctions can be updated
         Cancel the auction first

Error: "Not authorized to update this auction"
Solution: Only the seller (creator) can update/delete
*/
